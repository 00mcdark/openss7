\L{\LB{\Proc{AccessSelector}AccessSelector ::= \K{CHOICE} \{}}
\L{\LB{        entry [0] \K{NULL},}}
\L{\LB{                \C{}\-\- DUA identified by the entry}}

\CE{}\L{\LB{        prefix [3] NameList,}}
\L{\LB{                \C{}\-\- This identifies a prefix name for specified DUAs}}

\L{\LB{                \C{}\-\- For specifying group rights}}
\CE{}\L{\LB{        \}}}
